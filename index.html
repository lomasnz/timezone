=LET( sites, FILTER(Sites!A3:A, Sites!A3:A<>""), start, FILTER(Sites!E3:E, Sites!A3:A<>""), tStep, FILTER(Task_Template!A3:A, Task_Template!A3:A<>""), tTask, FILTER(Task_Template!B3:B, Task_Template!A3:A<>""), tOffset, FILTER(Task_Template!C3:C, Task_Template!A3:A<>""), tDuration, FILTER(Task_Template!D3:D, Task_Template!A3:A<>""), tScope, FILTER(Task_Template!F3:F, Task_Template!A3:A<>""), siteSteps, FILTER(tStep, tScope="SITE"), siteTasks, FILTER(tTask, tScope="SITE"), siteOffsets, FILTER(tOffset, tScope="SITE"), siteDurations, FILTER(tDuration, tScope="SITE"), globalSteps, IFERROR(FILTER(tStep, tScope="GLOBAL"), ""), globalTasks, IFERROR(FILTER(tTask, tScope="GLOBAL"), ""), globalOffsets, IFERROR(FILTER(tOffset, tScope="GLOBAL"), ""), globalDurations, IFERROR(FILTER(tDuration, tScope="GLOBAL"), ""), nSites, ROWS(sites), nTasks, ROWS(siteSteps), nRows, nSites * nTasks, taskIdx, MOD(SEQUENCE(nRows) - 1, nTasks) + 1, siteIdx, INT((SEQUENCE(nRows) - 1) / nTasks) + 1, siteData, SPLIT( FLATTEN( INDEX(siteSteps, taskIdx) & "♦" & INDEX(sites, siteIdx) & "♦" & INDEX(siteTasks, taskIdx) & "♦" & (INDEX(start, siteIdx) + INDEX(siteOffsets, taskIdx)) & "♦" & INDEX(siteDurations, taskIdx) ), "♦" ), globalData, IF( COUNTA(globalSteps)=0, , { globalSteps, ARRAYFORMULA(IF(globalSteps<>"","ALL","")), globalTasks, INDEX(start,1) + globalOffsets, globalDurations } ), SORT( VSTACK(siteData, globalData), 2, TRUE, 1, TRUE ) )
